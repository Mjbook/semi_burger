
CREATE TABLE BURGER_SHOP (
	SHOP_KEY NUMBER(10) PRIMARY KEY,
	SHOP_NAME VARCHAR2(30) NOT NULL,
	SHOP_ADDR VARCHAR2(100) NOT NULL,
	SHOP_TEL VARCHAR2(50) NOT NULL, 
	SHOP_OPEN DATE ,
	SHOP_CLOSE DATE
)

CREATE TABLE BURGER (
	ITEM_KEY NUMBER(10) PRIMARY KEY,
	ITEM_NUM NUMBER(10) NOT NULL,
	ITEM_NAME VARCHAR2(100) NOT NULL,
	ITEM_PAY VARCHAR2(100) NOT NULL,
	ITEM_COUNT NUMBER(20) NOT NULL,
	ITEM_EVENT_START DATE ,
	ITEM_EVENT_FINISH DATE ,
	ITEM_IMG_SRC VARCHAR2(100)
)

CREATE TABLE MEMBER (
	MEMBER_NO NUMBER(20) PRIMARY KEY,
	ID	VARCHAR2(30) NOT NULL,
	PASSWD VARCHAR2(30) NOT NULL,
	NAME VARCHAR2(20) NOT NULL,
	BIRTH VARCHAR2(10) NOT NULL,
	CELLPHONE VARCHAR2(11) NOT NULL,
	EMAIL VARCHAR2(30) NOT NULL,
	ADDRESS VARCHAR(200) NOT NULL,
	DISCOUNT NUMBER(20) ,
	GRADE VARCHAR2(20)
)

CREATE TABLE BOARD (
	NOTICE_NO NUMBER(22) PRIMARY KEY,
	NAME VARCHAR2(20) NOT NULL,
	SUBJECT VARCHAR2(300) NOT NULL,
	CONTENT VARCHAR2(4000) NOT NULL,
	INSERT_DATE DATE,
	BOARD_COUNT NUMBER(22)
)

CREATE TABLE QNA_BOARD (
	QNA_NO NUMBER(22) PRIMARY KEY,
	NAME VARCHAR2(20) NOT NULL,
	SUBJECT VARCHAR2(150) NOT NULL,
	CONTENT VARCHAR2(4000) NOT NULL,
	INSERT_DATE DATE ,
	BOARD_COUNT NUMBER(22),
	REF NUMBER(20) NOT NULL,
	LEV NUMBER(22) NOT NULL,
	SUNBUN NUMBER(22) NOT NULL
)

CREATE TABLE MY_KNOWHOW (
	KNOWHOW_NO NUMBER(22) PRIMARY KEY,
	NAME VARCHAR2(20) NOT NULL,
	SUBJECT VARCHAR2(150) NOT NULL,
	CONTENT VARCHAR2(4000) NOT NULL,
	INSERT_DATE DATE ,
	BOARD_COUNT NUMBER(22),
	MY_IMG_URL VARCHAR2(100)
)

CREATE TABLE ORDER_LIST (
	ORDER_NO NUMBER(20) PRIMARY KEY,
	ITEM_NAME VARCHAR2(100) NOT NULL,
	ITEM_COUNT NUMBER(10),
	ORDER_DATE DATE NOT NULL,
	ORDER_SHOP VARCHAR2(100) NOT NULL,
	ORDER_USER VARCHAR2(100) NOT NULL,
	TOTAL_PAY VARCHAR2(100) NOT NULL
)

CREATE TABLE MEMBER_BOARD_COUNT (
	MEMBER_IDX NUMBER(10) PRIMARY KEY,
	BOARD_IDX NUMBER(10) NOT NULL,
	COUNT_CHECK NUMBER(10) NOT NULL
)

CREATE TABLE TOPING (
	TOPING_KEY NUMBER(10) PRIMARY KEY,
	TOPING_NUM NUMBER(10) NOT NULL,
	TOPING_NAME VARCHAR2(100) NOT NULL,
	TOPING_PAY VARCHAR2(100) NOT NULL
)
